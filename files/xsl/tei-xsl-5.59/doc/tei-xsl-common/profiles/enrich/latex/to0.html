  <xsl:param name="before"/>
  <xsl:param name="style"/>
  <xsl:param name="after"/>
  <xsl:value-of select="$before"/>
  <xsl:choose>
    <xsl:when test="$style='italic'">
      <xsl:text>\textit{</xsl:text>
      <xsl:value-of select="normalize-space(.)"/>
      <xsl:text>}</xsl:text>
    </xsl:when>

      <xsl:text>\textbf{</xsl:text>

    </xsl:when>
    <xsl:otherwise>
      <xsl:value-of select="normalize-space(.)"/>
    </xsl:otherwise>
  </xsl:choose>
